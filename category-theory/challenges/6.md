2. Implement Shape, Circle, Rectangle, and area in Java

```
interface Shape {
    Float area();
}

class Circle implements Shape {
    Float r;

    constructor(Float r) {
        this.r = r;
    }

    Float area() {
        return 3.14 * r;
    }
}

class Rectangle implements Rectangle {
    Float x, y;

    constructor(Float x, Float y) {
        this.x = x;
        this.y = y;
    }

    Float area() {
        return this.x * this.y;
    }
}

Float area(s: Shape) {
    return s.area();
}
```
